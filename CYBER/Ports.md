- from 0 to 65535
- 0 - 1024: common port
	- 0: reserved
	- 21: FTP
	- 22: SSH
	- 80: HTTP
	- 443: HTTPS
	- 445: SMB
	- more: https://www.vmaxx.net/techinfo/ports.htm
- 3389: Remote Desktop Protocol
